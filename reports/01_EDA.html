df = X.copy()

            'household_type', 'annual_income_range']

fig, axes = plt.subplots(1, 5, figsize=(20, 4))
for ax, col in zip(axes.flat, cat_cols):
    rates = df.groupby(col)['became_chronic'].mean().sort_values(ascending=True)
